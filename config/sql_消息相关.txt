		((from_id='oCBrx0FreB-L8pIQM5_RYDGoWOJJ' AND to_id='oCBrx0FreB-L8pIQM5_RYDGoWOKQ') OR
		(from_id='oCBrx0FreB-L8pIQM5_RYDGoWOKQ' AND to_id='oCBrx0FreB-L8pIQM5_RYDGoWOJJ')) ORDER BY TIME DESC LIMIT 0,10)AS a
